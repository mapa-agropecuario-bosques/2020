Rasterización

Introducción

Este documento computacional rasteriza capas vectoriales.

Código para la carga de paquetes 
# Paquetes
library(here)
library(dplyr)
library(sf)
library(terra)

  st_make_valid()

# Recorte del objeto sf de Costa Rica
costarica_sf <- 
  costarica_sf |>
  st_crop(st_bbox(
    c(
      xmin = 280000, 
      ymin = 880000, 
      xmax = 660000, 
      ymax = 1250000
    ), 
    crs = st_crs(costarica_sf)
  ))

  st_read(ARCHIVO_VECTORIAL_ASP, quiet = TRUE) |>
  filter(cat_manejo == "Parque Nacional" | cat_manejo == "Reserva Biologica" | cat_manejo == "Monumento Natural") |>
  st_transform(5367) |>
  st_make_valid()

# Objeto raster de ASP
capa_terra <-
  rasterize(capa_sf, costarica_terra)

# Asignación de CRS
crs(capa_terra) <- "EPSG:5367"

# Recorte
capa_terra <-
  capa_terra |>
  crop(costarica_sf) |>
  mask(costarica_sf)
